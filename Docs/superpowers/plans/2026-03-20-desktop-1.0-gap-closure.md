# Desktop 1.0 Gap Closure — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Close all PRD v1.6 gaps in the desktop runtime UI, bringing it from ~70% to 1.0-complete across 6 waves of work.

**Architecture:** Each wave targets one subsystem. Waves are ordered by product impact — scene interaction first (makes the product usable), then layout polish, then permissions, then employee management, then editor wiring, then integration fixes. Each wave produces a working, testable state.

**Tech Stack:** React 19, Three.js (react-three-fiber), SVG, Tailwind CSS, shadcn/ui, PixiJS 8 (hidden backing renderer), GSAP 3, localStorage for UI preferences. All work is in `packages/ui-office/` unless noted.

**Key Constraints:**
- No Framer Motion (PixiJS + GSAP for scene, CSS transitions for DOM)
- Desktop is 1.0 reference environment
- Follow existing event patterns (`packages/shared-types/src/events.ts`)
- Reuse existing services before adding abstractions

- [ ] **Step 1: Create PerformanceHUD component**

```tsx
// src/components/scene/PerformanceHUD.tsx
import { useEffect, useRef, useState } from 'react';

interface PerformanceStats {
  fps: number;
  frameTime: number;
  heapUsed: number;  // MB
  heapTotal: number; // MB
}

export function PerformanceHUD() {
  const [stats, setStats] = useState<PerformanceStats>({ fps: 0, frameTime: 0, heapUsed: 0, heapTotal: 0 });
  const [visible, setVisible] = useState(() => {
    try { return localStorage.getItem('offisim.perf.hud') === 'true'; } catch { return false; }
  });
  const framesRef = useRef(0);
  const lastTimeRef = useRef(performance.now());

  useEffect(() => {
    if (!visible) return;
    let raf: number;
    const tick = () => {
      framesRef.current++;
      const now = performance.now();
      const elapsed = now - lastTimeRef.current;
      if (elapsed >= 1000) {
        const fps = Math.round((framesRef.current * 1000) / elapsed);
        const frameTime = +(elapsed / framesRef.current).toFixed(1);
        const mem = (performance as any).memory;
        setStats({
          fps,
          frameTime,
          heapUsed: mem ? +(mem.usedJSHeapSize / 1048576).toFixed(0) : 0,
          heapTotal: mem ? +(mem.totalJSHeapSize / 1048576).toFixed(0) : 0,
        });
        framesRef.current = 0;
        lastTimeRef.current = now;
      }
      raf = requestAnimationFrame(tick);
    };
    raf = requestAnimationFrame(tick);
    return () => cancelAnimationFrame(raf);
  }, [visible]);

  useEffect(() => {
    const handler = (e: KeyboardEvent) => {
      if (e.key === 'F2') {
        setVisible((v) => {
          const next = !v;
          try { localStorage.setItem('offisim.perf.hud', String(next)); } catch {}
          return next;
        });
      }
    };
    window.addEventListener('keydown', handler);
    return () => window.removeEventListener('keydown', handler);
  }, []);

  if (!visible) return null;

  return (
    <div className="absolute top-2 left-2 z-40 pointer-events-none font-mono text-[10px] leading-tight text-green-400/80 bg-black/60 rounded-lg px-2 py-1.5 backdrop-blur-sm border border-green-500/20">
      <div>{stats.fps} FPS ({stats.frameTime}ms)</div>
      {stats.heapUsed > 0 && <div>MEM {stats.heapUsed}/{stats.heapTotal} MB</div>}
      <div className="text-green-500/40 mt-0.5">F2 toggle</div>
    </div>
  );
}
```

```tsx
// src/components/employees/SkillBindingList.tsx
import { useAicsRuntime } from '@aics/ui-office/runtime';
import { useEffect, useState } from 'react';
import { Puzzle, ExternalLink } from 'lucide-react';
import { Badge } from '@aics/ui-core/badge';

interface InstalledSkill {
  asset_id: string;
  package_id: string;
  kind: string;
  label: string;
  version: string;
}

export function SkillBindingList({ employeeId }: { employeeId: string }) {
  const { repos } = useAicsRuntime();
  const [skills, setSkills] = useState<InstalledSkill[]>([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    if (!repos?.installedAssets) { setLoading(false); return; }
    repos.installedAssets.findByEmployee?.(employeeId)
      .then((assets) => {
        setSkills(assets.filter((a) => a.kind === 'skill').map((a) => ({
          asset_id: a.asset_id,
          package_id: a.package_id,
          kind: a.kind,
          label: a.label || a.asset_id,
          version: a.version || '0.0.0',
        })));
      })
      .catch(() => setSkills([]))
      .finally(() => setLoading(false));
  }, [repos, employeeId]);

  if (loading) return <div className="text-[10px] text-slate-500">Loading skills...</div>;

  return (
    <div className="space-y-2">
      <div className="flex items-center justify-between">
        <label className="text-xs font-medium text-slate-300">Installed Skills</label>
        <span className="text-[10px] text-slate-600">{skills.length} bound</span>
      </div>
      {skills.length === 0 ? (
        <div className="text-center py-3 text-[10px] text-slate-600">
          <Puzzle className="h-5 w-5 mx-auto mb-1 opacity-30" />
          <p>No skills installed</p>
          <p className="opacity-60 mt-0.5">Install skill packages from the marketplace or import SKILL.md files</p>
        </div>
      ) : (
        <div className="space-y-1.5">
          {skills.map((s) => (
            <div key={s.asset_id} className="flex items-center justify-between rounded-lg border border-white/5 bg-black/20 px-2.5 py-1.5">
              <div className="flex items-center gap-2">
                <Puzzle className="h-3 w-3 text-blue-400" />
                <span className="text-xs text-white/80">{s.label}</span>
              </div>
              <Badge variant="secondary" className="text-[9px]">v{s.version}</Badge>
            </div>
          ))}
        </div>
      )}
    </div>
  );
}
```
